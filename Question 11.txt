SELECT DeptID, COUNT(*) AS TotalStudents
FROM Student
GROUP BY DeptID;


SELECT DeptID, COUNT(*) AS TotalStudents
FROM Student
GROUP BY DeptID
HAVING COUNT(*) > 2;

CREATE OR REPLACE VIEW StudentDepartment AS SELECT s.StudentID, s.Name, d.DeptName FROM Student s JOIN Department d ON s.DeptID = d.DeptID;


SELECT * FROM StudentDepartment;
